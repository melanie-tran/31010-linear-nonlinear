tol <- 10^(-10)

x <- NULL
f <- function(x){ 
  matrix(c((x[1] + x[2] + x[3]**2 -12), 
            (x[1]**2 -x[2] + x[3] - 2), 
            (2*x[1]**2 -x[2]**2 + x[3] - 1)),byrow=TRUE,nrow=length(x))
}
Df <- function(x){
  (matrix(c(1, 1, 2*x[3], 
            2*x[1],-1,1,
            4*x[1], -2*x[2],1), byrow = TRUE, nrow=length(x)))
}

while (err>tol){
